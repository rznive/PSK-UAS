create table
  public.activity_report (
    id bigint generated by default as identity not null,
    nama_gunung character varying null,
    ttl_report character varying null,
    status_gunung character varying null,
    desc_report character varying null,
    report_by character varying null,
    detail_url character varying null,
    created_at timestamp with time zone not null default now(),
    image_url character varying null,
    periode character varying null,
    constraint activity_report_pkey primary key (id)
  ) tablespace pg_default;

  create table
  public.data_commentar (
    id bigint generated by default as identity not null,
    activity_report_id bigint not null,
    user_id uuid not null,
    comment_text character varying not null,
    created_at timestamp with time zone not null default now(),
    constraint data_commentar_pkey primary key (id)
  ) tablespace pg_default;


  create table
  public.user_login (
    id uuid not null default gen_random_uuid (),
    nama character varying null,
    email character varying null,
    password character varying null,
    created_at timestamp with time zone not null default now(),
    constraint user_login_pkey primary key (id),
    constraint user_login_email_key unique (email)
  ) tablespace pg_default;
